@subsidy_minimum = -0.75
@subsidy_maximum = 0.75

# these values are specific to these agreement_term_values, but are based on the _basic and _advanced values (which are equal as of 3.6 "Orion")
@subsidy_multiplier_mercatorum = 0.5
@subsidy_monthly_multiplier_mercatorum = 5

specialist_mercatorum = {
	term = specialist_type
	triggered_loyalty_change = {
		trigger = { target = { is_xenophile = yes } }
		loyalty_change = 5
	}
	triggered_loyalty_change = {
		trigger = { target = { is_capitalist = yes } }
		loyalty_change = 5
	}
	triggered_loyalty_change = {
		trigger = { target = { is_oligarchy = yes } }
		loyalty_change = 5
	}
	triggered_loyalty_change = {
		trigger = { target = { is_xenophobe = yes } }
		loyalty_change = -20
	}
	activate_effect = {
		convert_to_specialist = specialist_mercatorum
	}
	potential = {
		agreement_preset = preset_mercatorum
	}
	possible = {
		owner = { NOT = { has_specialist_subject_type = { TYPE = mercatorum } } }
		target = {
			is_machine_empire = no
			is_hive_empire = no
		}
	}
}
